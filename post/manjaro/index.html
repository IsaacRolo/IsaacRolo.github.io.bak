2

sudo pacman -Syy
sudo pacman-mirrors -i -c China -m rank  #选一个清华源就行
sudo pacman -Syyu

添加arch源

编辑/etc/pacman.conf文件

1

[archlinuxcn]SigLevel = Optional TrustedOnlyServer = https://mirrors.sjtug.sjtu.edu.cn/archlinux-cn/$arch
1

sudo pacman -Syy && sudo pacman -S archlinuxcn-keyring

升级系统

1

sudo pacman -Su

输入法

1
2
3

sudo pacman -S fcitx-sogoupinyin
sudo pacman -S fcitx-im # 全部安装
sudo pacman -S fcitx-configtool # 图形化配置工具

设置中文输入法环境变量，编辑~/.xprofile文件，增加下面几行(如果文件不存在，则新建)

1
2
3

export GTK_IM_MODULE=fcitx
export QT_IM_MODULE=fcitx
export XMODIFIERS="@im=fcitx"

安装Docker

1
2
3
4
5
6
7
8

# Pacman 安装 Docker
sudo pacman -S docker
# 启动docker服务
sudo systemctl start docker 
# 查看docker服务的状态
sudo systemctl status docker
# 设置docker开机启动服务
sudo systemctl enable docker

配置docker镜像源 sudo nano /etc/docker/daemon.json 加入 ”registry-mirrors”: [“http://xxx.com”]

https://docker.mirrors.ustc.edu.cn https://registry.docker-cn.com http://hub-mirror.c.163.com https://pee6w651.mirror.aliyuncs.com
